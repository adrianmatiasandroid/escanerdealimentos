        <script src="https://cdnjs.cloudflare.com/ajax/libs/tesseract.js/2.1.1/tesseract.min.js"></script>

        <script>

            let videoElement = document.getElementById('video');

            let captureButton = document.getElementById('captureButton');

            let resultContainer = document.getElementById('result');

            let errorMessage = document.getElementById('error-message');

            let ocrBoundingBoxes = [];

            // Acceso a la cámara

            async function startCamera() {

                try {

                    const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' } });

                    videoElement.srcObject = stream;

                } catch (err) {

                    errorMessage.textContent = "Error al acceder a la cámara. Asegúrese de tener los permisos adecuados.";

                    console.error("Camera access error: ", err);

                }

            }

            // Tomar captura de la imagen

            captureButton.addEventListener('click', async () => {

                errorMessage.textContent = '';

                let canvas = document.createElement('canvas');

                canvas.width = videoElement.videoWidth;

                canvas.height = videoElement.videoHeight;

                let ctx = canvas.getContext('2d');

                ctx.drawImage(videoElement, 0, 0, canvas.width, canvas.height);

                // Usar Tesseract.js para OCR

                const { data: { text, blocks } } = await Tesseract.recognize(

                    canvas,

                    'spa',

                    {

                        logger: (m) => console.log(m),

                    }

                );

                // Mostrar los resultados y dibujar cuadros delimitadores

                resultContainer.innerHTML = `<h2>Ingredientes Reconocidos:</h2><pre>${text}</pre>`;

                drawBoundingBoxes(blocks);

                analyzeIngredients(text);

            });

            // Dibujar los cuadros delimitadores para los ingredientes detectados

            function drawBoundingBoxes(blocks) {

                ocrBoundingBoxes.forEach(box => box.remove());

                ocrBoundingBoxes = [];

                blocks.forEach(block => {

                    if (block.text.trim()) {

                        let box = document.createElement('div');

                        box.classList.add('ocr-box');

                        box.style.left = `${block.bbox[0]}px`;

                        box.style.top = `${block.bbox[1]}px`;

                        box.style.width = `${block.bbox[2] - block.bbox[0]}px`;

                        box.style.height = `${block.bbox[3] - block.bbox[1]}px`;

                        document.body.appendChild(box);

                        ocrBoundingBoxes.push(box);

                    }

                });

            }

            // Análisis de ingredientes

            async function analyzeIngredients(text) {

                let ingredientsList = text.split('\n').filter(line => line.trim() !== '');

                let resultHTML = '';

                for (let ingredient of ingredientsList) {

                    // Llamar a la API de Open Food Facts para obtener información sobre el ingrediente

                    let response = await fetch(`https://world.openfoodfacts.org/api/v0/product/ingredients.json?ingredients=${ingredient}`);

                    let data = await response.json();

                    if (data && data.ingredients) {

                        let ingredientData = data.ingredients[0];

                        let vegan = ingredientData.vegetarian || 'Desconocido';

                        let healthyScore = ingredientData.nutriments ? ingredientData.nutriments["energy-kcal_100g"] : 0;

                        resultHTML += `

                            <div class="ingredient">

                                <strong>${ingredient}</strong>

                                <p class="${vegan.toLowerCase() === 'yes' ? 'vegan' : 'non-vegan'}">Vegan: ${vegan}</p>

                                <p>Descripción: ${ingredientData.text}</p>

                                <p class="healthy">Puntuación de Salud: ${healthyScore ? 10 - healthyScore : 'N/A'}</p>

                            </div>

                        `;

                    } else {

                        resultHTML += `<p>Ingrediente no encontrado: ${ingredient}</p>`;

                    }

                }

                resultContainer.innerHTML = resultHTML;

            }

            // Iniciar cámara al cargar la página

            window.addEventListener('load', startCamera);

        </script>
